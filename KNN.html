In[3]:
# opening and loading the Dataset
file_path = 'E:/cc_transctn/credit_card_transactions.csv'
credit_card_data = pd.read_csv(file_path)
In[4]:
# performing Data preprocessing and necessary Data cleaning
columns_to_drop = ['Unnamed: 0', 'trans_date_trans_time', 'cc_num', 'first', 'last', 'street', 'trans_num', 'dob']
credit_card_data_cleaned = credit_card_data.drop(columns=columns_to_drop)

credit_card_data_cleaned.fillna({'merch_zipcode': credit_card_data_cleaned['merch_zipcode'].mean()}, inplace=True)
In[5]:
# Encoding categorical variable columns
categorical_columns = ['merchant', 'category', 'gender', 'city', 'state', 'job']
label_encoders = {col: LabelEncoder() for col in categorical_columns}
for col in categorical_columns:
    credit_card_data_cleaned[col] = label_encoders[col].fit_transform(credit_card_data_cleaned[col])
In[6]:
# Defining of featue variables and target variables
X =  credit_card_data_cleaned.drop('is_fraud', axis=1)
Y =  credit_card_data_cleaned['is_fraud']
# Split the data into training and testing sets (20% test, 80% train)
X_train, X_test, Y_train, Y_test = train_test_split(X, Y, test_size=0.2, random_state=42, stratify=Y)
In[7]:
# Appling SMOTE to balance the dataset
smote = SMOTE(random_state=42)
X_train_resampled, Y_train_resampled = smote.fit_resample(X_train, Y_train)

# Standardizing the features
scaler = StandardScaler()
X_train_resampled = scaler.fit_transform(X_train_resampled)
X_test = scaler.transform(X_test)
In[8]:
# K-Nearest Neighbors model
knn = KNeighborsClassifier(n_neighbors=5)

# Cross-validation to evaluate model performance
cv_scores = cross_val_score(knn, X_train_resampled, Y_train_resampled, cv=5, scoring='precision')

# fitting the datas and training the KNN model
knn.fit(X_train_resampled, Y_train_resampled)

# Predictions
Y_pred = knn.predict(X_test)
In[9]:
# Evaluate model performance
precision = precision_score(Y_test, Y_pred)
classification_rep = classification_report(Y_test, Y_pred)
conf_matrix = confusion_matrix(Y_test, Y_pred)
roc_auc = roc_auc_score(Y_test, Y_pred)
In[10]:
